export function createRng(seed = 123456789) {
  let s = seed >>> 0;
  return {
    next() {
      // xorshift32
      s ^= s << 13;
      s ^= s >>> 17;
      s ^= s << 5;
      return (s >>> 0) / 0xffffffff;
    },
    roll(max = 6) {
      return 1 + Math.floor(this.next() * max);
    },
  };
}
